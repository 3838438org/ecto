461  template<typename Impl>
462  bool cell_<Impl>::init()
463  {
464  try {
465  if(!impl_)
466  {
467  impl_.reset(new Impl);
468  Impl* i=impl_.get();
469  //these handle finalizing the registration of spores that
470  //were registered at static time.
471  parameters.realize_potential(i);
472  inputs.realize_potential(i);
473  outputs.realize_potential(i);
474  }
475  return impl_;
476  } catch (const std::exception& e) {
477  ECTO_TRACE_EXCEPTION("const std::exception&");
478  BOOST_THROW_EXCEPTION(except::CellException()
479  << except::when("Construction")
480  << except::type(name_of(typeid(e)))
481  << except::cell_name(name())
482  << except::what(e.what()));
483  } catch (...) {
484  ECTO_TRACE_EXCEPTION("...");
485  BOOST_THROW_EXCEPTION(except::CellException()
486  << except::when("Construction")
487  << except::what("(unknown exception)")
488  << except::cell_name(name()));
489  }
